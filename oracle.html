    // 🔮 Listen for messages from Thunkable

    window.addEventListener("message", function(event) {

      try {

        // Support both JSON payloads and plain text

        let data = event.data;

        if (typeof data === "string") {

          try { data = JSON.parse(data); } catch {}

        }


        const message = data.html || data.message || "(No message received)";

        const oracleDiv = document.getElementById("oracleResponse");

        oracleDiv.innerHTML = message;

        console.log("✅ Oracle HTML updated with new message:", message);


        // Send confirmation back to Thunkable

        window.parent.postMessage("oracleLoaded", "*");

      } catch (err) {

        console.error("⚠️ Error processing message:", err);

      }

    });
